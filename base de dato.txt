CREATE TABLE client(
	id INTEGER NOT NULL,
	name TEXT,
	PRIMARY KEY(id)
);

CREATE TABLE showcase(
	id INTEGER NOT NULL,
	movie TEXT NOT NULL,
	day INT NOT NULL,
	room INT NOT NULL,
	PRIMARY KEY(id)
);

CREATE TABLE booking(
	client_id INTEGER NOT NULL,
	showcase_id INTEGER NOT NULL,
	cancelled INTEGER NOT NULL,
	seat INTEGER NOT NULL,
	FOREIGN KEY (client_id) REFERENCES client(id),
	FOREIGN KEY (showcase_id) REFERENCES showcase(id)
);


EN SQLITE, INTEGER PRIMARY KEY SE INCREMENTA SOLO, ENTONCES
SE PUEDE INSERTAR EN LA TABLA E IGNORAR PONER ID Y SE LLENA SOLO.
